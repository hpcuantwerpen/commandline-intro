\def\trapezebg#1% trailing comments needed to avoid inserting empty lines/spaces before the graphic?
              {\setupMPvariables[Trapeze][#1]%
                \startuseMPgraphic{Trapeze}
                  StartPage;
                  fill Page withcolor \MPcolor{vscbackground};
                  % Slope of trapeze should be 12 degrees -> tan(12)=0.213
                  filldraw((0cm,0cm)
                    -- (PaperWidth,0cm)
                    -- (PaperWidth,\MPvar{height}+0.213*PaperWidth)
                    -- (0cm,\MPvar{height}) -- cycle)
                    withcolor \MPcolor{vscorange};
                  StopPage;
                \stopuseMPgraphic
                \useMPgraphic{Trapeze}}
